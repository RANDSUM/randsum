Getting Started

Installation

npm npm install @randsum/dice
yarn yarn add @randsum/dice
bun bun add @randsum/dice

Quick Start

import { roll, D20, D6 } from '@randsum/dice'

// Simple rolls
const d20Roll = D20.roll()
const d6Roll = roll(6)

// Dice notation
const advantageRoll = roll('2d20H') // Roll 2d20, keep highest
const statRoll = roll('4d6L')       // Roll 4d6, drop lowest

// Complex modifiers
const complexRoll = roll('3d8!R{<3}+5')
// 3d8, explode on max, reroll less than 3, add 5

console.log('D20:', d20Roll.total)
console.log('Advantage:', advantageRoll.total)
console.log('Stat roll:', statRoll.total)
console.log('Complex:', complexRoll.total)

Basic D&D Rolls

import { roll } from '@randsum/dice'

// Attack roll with advantage
const attack = roll('1d20+5', '2d20H+5')
console.log('Attack:', attack.total)

// Damage roll
const damage = roll('1d8+3')
console.log('Damage:', damage.total)

// Ability score generation
const stats = [
  roll('4d6L'), roll('4d6L'), roll('4d6L'),
  roll('4d6L'), roll('4d6L'), roll('4d6L')
]
console.log('Stats:', stats.map(s => s.total))

Game System Examples

import { roll5e } from '@randsum/5e'
import { rollBlades } from '@randsum/blades'
import { rollDH } from '@randsum/daggerheart'

// D&D 5e skill check with advantage
const skillCheck = roll5e({
  modifier: 5,
  rollingWith: 'Advantage'
})
console.log('5e Check:', skillCheck.total)

// Blades in the Dark action roll
const actionRoll = rollBlades(3) // 3 dice pool
console.log('Blades Result:', actionRoll.result)

// Daggerheart Hope/Fear roll
const dhRoll = rollDH({ modifier: 2 })
console.log('Daggerheart:', dhRoll.total, dhRoll.type)

Advanced Modifiers

import { roll } from '@randsum/dice'

// Exploding dice (reroll max values)
const exploding = roll('3d6!')
console.log('Exploding:', exploding.total)

// Reroll low values
const reroll = roll('4d6R{<3}')
console.log('Reroll low:', reroll.total)

// Complex combination
const complex = roll('2d10!R{<3}+5')
console.log('Complex:', complex.total)

// Multiple modifiers
const ultimate = roll('4d8!R{<2}L+3')
// 4d8, explode on 8, reroll <2, drop lowest, +3
console.log('Ultimate:', ultimate.total)
